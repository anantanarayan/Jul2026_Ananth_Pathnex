## Ansible - Handler to restart Nginx

---
    name : Manage Nginx 
    hosts : all 
    become : yes

    tasks: 
        -name : Update config file 
        copy :
           src : index.html
           dest : /usr/share/nginx/index.html
        notify : Restart Nginx

    handlers :
        -name : Restart Nginx 
        service :
          name : nginx
          state : restarted

#jenkins pipelline - use envitrnonment Variables 

how to use environment variables in jenkins pipelines **

pipeline {
    agent any {
        environment  {
            INST = "PATHNE"
        }
    }
    stages {
        stage ('print'){
            steps {
                sh 'echo "welcome to $INST Devops Training " '
            }
        }
        stage ('Build') {
            steps {
                sh 'mvn clean package -Dinstitute.name=$INST'
            }
        }
    }
        }
    }
}
